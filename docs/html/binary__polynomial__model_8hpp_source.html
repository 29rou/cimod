330  void add_linear(const IndexType &v, const FloatType &bias, const Vartype vartype = Vartype::NONE) {
331 
332  //Check vartype
333  if (vartype != Vartype::NONE) {
334  if ((m_vartype == Vartype::SPIN) && (vartype == Vartype::BINARY)) {
335  std::cerr << "Cannot convert the vartype=SPIN to vartype=BINARY" << std::endl;
336  }
337  else if((m_vartype == Vartype::BINARY) && (vartype == Vartype::SPIN)) {
338  std::cerr << "Cannot convert the vartype=BINARY to vartype=SPIN" << std::endl;
339  }
340  else {
341  std::cerr << "Unknown vartype" << std::endl;
342  }
343  }
344 
345  // Insert or assign the bias
346  FloatType value = 0;
347  std::vector<IndexType> index = {v};
348  if (m_polynomial.count(index) != 0) {
349  value = m_polynomial[index];
350  }
351  insert_or_assign(m_polynomial, index, value + bias);
352  m_variables.emplace(v);
353  }

360  void add_interaction(const std::vector<IndexType> &u, const FloatType &bias, const Vartype vartype = Vartype::NONE) {
361 
362  //Check the input interaction is valid
363  for (const auto &it : u) {
364  if (std::count(u.begin(), u.end(), it) != 1) {
365  std::cerr << "No self-loops allowed, therefore (";
366  for (const auto &it_print : u) {
367  std::cerr << it_print << ", ";
368  }
369  std::cerr << ") is not an allowed interaction" << std::endl;
370  return;
371  }
372  }
373 
374  //Check vartype
375  if((vartype != Vartype::NONE) && (vartype != m_vartype)) {
376  if((m_vartype == Vartype::SPIN) && (vartype == Vartype::BINARY)) {
377  std::cerr << "Cannot convert vartype=SPIN to vartype=BINARY" << std::endl;
378  return;
379  }
380  else if((m_vartype == Vartype::BINARY) && (vartype == Vartype::SPIN)) {
381  std::cerr << "Cannot convert vartype=BINARY to vartype=SPIN" << std::endl;
382  return;
383  }
384  else {
385  std::cerr << "Unknown vartype" << std::endl;
386  return;
387  }
388  }
389 
390  FloatType value = 0;
391  if (m_polynomial.count(u) != 0) {
392  value = m_polynomial[u];
393  }
394  insert_or_assign(m_polynomial, u, value + bias);
395 
396 
397  if (u.size() > 1) {
398  update_adjacency(u);
399  }
400 
401  for (const auto &it: u) {
402  if (m_variables.count(it) == 0) {
403  m_variables.emplace(it);
404  add_linear(it, 0);
405  }
406  }

478  void normalize(const std::pair<FloatType, FloatType> &bias_range = {1.0, 1.0},
479  const std::vector<std::vector<IndexType>> &ignored_variables = {}
480  ) {
481 
482  auto comp = [](const auto &a, const auto &b) { return a.second < b.second; };
483  auto min = std::min_element(m_polynomial.begin(), m_polynomial.end(), comp)->second/bias_range.first;
484  auto max = std::max_element(m_polynomial.begin(), m_polynomial.end(), comp)->second/bias_range.second;
485 
486  FloatType inv_scale = (min < max) ? max : min;
487 
488  if (inv_scale != 0.0) {
489  scale(1.0/inv_scale, ignored_variables);
490  for (const auto &it_polynomial: m_polynomial) {
491  if (it_polynomial.first.size() > 1) {update_adjacency(it_polynomial.first); };
492  }
493  }

508  FloatType energy(const Sample<IndexType> &sample) const {
509  FloatType en = 0.0;
510  for (const auto &it_polynomial: m_polynomial) {
511  int32_t multiple_variable = 1;
512  for (const auto &it_vec: it_polynomial.first) {
513  if (check_vartype(sample.at(it_vec), m_vartype)) {
514  multiple_variable *= sample.at(it_vec);
515  }
516  else {
517  multiple_variable = 0;
518  break;
519  }
520  }
521  en += static_cast<FloatType>(multiple_variable)*it_polynomial.second;
522  }
523  return en;
524  }
