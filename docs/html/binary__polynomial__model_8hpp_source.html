199  BinaryPolynomialModel(const std::vector<IndexType> &variables,
200  const PolynomialKeyList<std::size_t> &poly_key_distance_list,
201  const PolynomialValueList<FloatType> &poly_value_list,
202  const Vartype vartype
203  ): vartype_(vartype) {
204 
205  if (poly_key_distance_list.size() != poly_value_list.size()) {
206  throw std::runtime_error("The sizes of key_list and value_list must match each other");
207  }
208 
209  variables_ = std::unordered_set<IndexType>(variables.begin(), variables.end());
210 
211  if (variables_.size() != variables.size()) {
212  throw std::runtime_error("Unknown error. It seems that the input variables contain the same variables");
213  }
214 
215  std::size_t num_interactions = poly_key_distance_list.size();
216  poly_key_list_.resize(num_interactions);

219 #pragma omp parallel for
220  for (std::size_t i = 0; i < num_interactions; ++i) {
221  std::vector<IndexType> temp;
222  for (const auto &it: poly_key_distance_list[i]) {
223  temp.push_back(variables[it]);

225  std::sort(temp.begin(), temp.end());
226  poly_key_list_[i] = temp;
227  poly_value_list_[i] = poly_value_list[i];

231  poly_key_inv_[poly_key_list_[i]] = i;
232  for (const auto &it: poly_key_list_[i]) {

407  void remove_interaction(std::vector<IndexType> &key) {
408  std::sort(key.begin(), key.end());
409  if (poly_key_inv_.count(key) == 0) {
410  return;
411  }
412 
413  for (const auto &index: key) {
414  if (each_variable_num_[index] >= 2) {
415  each_variable_num_[index]--;
416  }
417  else if (each_variable_num_[index] == 1) {
418  each_variable_num_.erase(index);
419  variables_.erase(index);
420  relabel_flag_for_variables_to_integers_ = true;
421  }
422  }
423 
424  std::size_t inv = poly_key_inv_[key];
425 
426  std::swap(poly_key_inv_[key], poly_key_inv_[poly_key_list_.back()]);
427  poly_key_inv_.erase(key);
428 
429  std::swap(poly_key_list_[inv], poly_key_list_.back());
430  poly_key_list_.pop_back();
431 
432  std::swap(poly_value_list_[inv], poly_value_list_.back());
433  poly_value_list_.pop_back();

487  void add_interaction(std::vector<IndexType> &key, const FloatType &value, const Vartype vartype = Vartype::NONE) {
488  if (std::abs(value) <= 0.0) {
489  return;
490  }
491  std::sort(key.begin(), key.end());
492  CheckKeySelfLoop(key);
493  if (vartype_ == vartype || vartype == Vartype::NONE) {
494  SetKeyAndValue(key, value);
495  }
496  else {
497  const std::size_t original_key_size = key.size();
498  const std::size_t changed_key_list_size = IntegerPower(2, original_key_size);
499 
500  if (vartype_ == Vartype::SPIN && vartype == Vartype::BINARY) {
501  for (std::size_t i = 0; i < changed_key_list_size; ++i) {
502  const auto changed_key = GenerateChangedKey(key, i);
503  int sign = ((original_key_size - changed_key.size())%2 == 0) ? 1.0 : -1.0;
504  SetKeyAndValue(changed_key, value*IntegerPower(2, changed_key.size())*sign);
505  }
506  }
507  else if (vartype_ == Vartype::BINARY && vartype == Vartype::SPIN) {
508  FloatType changed_value = value*(1.0/changed_key_list_size);
509  for (std::size_t i = 0; i < changed_key_list_size; ++i) {
510  SetKeyAndValue(GenerateChangedKey(key, i), changed_value);
511  }
512  }
513  else {
514  throw std::runtime_error("Unknown vartype error");
515  }
516  }
517  }

575  FloatType energy(const Sample<IndexType> &sample, bool omp_flag = true) const {
576  if (sample.size() != get_num_variables()) {
577  throw std::runtime_error("The size of sample must be equal to num_variables");
578  }
579 
580  if (get_num_interactions() == 0) {
581  return 0.0;
582  }
583 
584  std::size_t num_interactions = get_num_interactions();
585  FloatType val = 0.0;
586 
587  if (omp_flag) {
588 #pragma omp parallel for reduction (+: val)
589  for (std::size_t i = 0; i < num_interactions; ++i) {
590  int32_t spin_multiple = 1;
591  for (const auto &index: poly_key_list_[i]) {
592  spin_multiple *= sample.at(index);
593  if (spin_multiple == 0.0) {
594  break;
595  }
596  }
597  val += spin_multiple*poly_value_list_[i];
598  }
599  }
600  else {
601  for (std::size_t i = 0; i < num_interactions; ++i) {
602  int32_t spin_multiple = 1;
603  for (const auto &index: poly_key_list_[i]) {
604  spin_multiple *= sample.at(index);
605  if (spin_multiple == 0.0) {
606  break;
607  }
608  }
609  val += spin_multiple*poly_value_list_[i];
610  }
611  }
612  return val;
613  }

710  poly_value_list_[poly_key_inv_[std::vector<IndexType>{}]] *= scalar_inv;

718  void normalize(const std::pair<FloatType, FloatType> &range = {1.0, 1.0},
719  const PolynomialKeyList<IndexType> &ignored_interactions = {},
720  const bool ignored_offset = false) {
721 
722  if (get_num_interactions() == 0) {
723  return;
724  }
725 
726  FloatType max_poly_value = poly_value_list_[0];
727  FloatType min_poly_value = poly_value_list_[0];
728 
729  for (const auto &poly_value: poly_value_list_) {
730  if (max_poly_value < poly_value) {
731  max_poly_value = poly_value;
732  }
733  if (min_poly_value > poly_value) {
734  min_poly_value = poly_value;
735  }
736  }
737 
738  FloatType inv_scale = std::max(min_poly_value/range.first, max_poly_value/range.second);
739 
740  if (inv_scale != 0.0) {
741  scale(1.0/inv_scale, ignored_interactions, ignored_offset);
742  }
743  }

854  nlohmann::json to_serializable() const {
855  nlohmann::json output;
856  if (vartype_ == Vartype::BINARY) {

859  else if (vartype_ == Vartype::SPIN) {

866  std::size_t num_interactions = get_num_interactions();
867  PolynomialKeyList<std::size_t> poly_key_distance_list(num_interactions);
868  std::vector<IndexType> sorted_variables = get_sorted_variables();
869 
870 #pragma omp parallel for
871  for (std::size_t i = 0; i < num_interactions; ++i) {
872  std::vector<std::size_t> temp;
873  for (const auto &it: poly_key_list_[i]) {
874  auto it_index = std::lower_bound(sorted_variables.begin(), sorted_variables.end(), it);
