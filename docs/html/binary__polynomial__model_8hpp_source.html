286  void add_variable(const IndexType &v, const FloatType &bias, const Vartype vartype = Vartype::NONE) {
287 
288  FloatType b = bias;
289 
290  // handle the case that a different vartype is provided
291  if((vartype != Vartype::NONE) && (vartype != m_vartype)) {
292  if((m_vartype == Vartype::SPIN) && (vartype == Vartype::BINARY)) {
293  b /= 2;
294  m_offset += b;
295  }
296  else if((m_vartype == Vartype::BINARY) && (vartype == Vartype::SPIN)) {
297  m_offset -= b;
298  b *= 2;
299  }
300  else {
301  std::cerr << "Unknown vartype" << std::endl;
302  }
303  }

323 
324  //Check the input interaction is valid
325  for (auto &it : u) {
326  if (std::count(u.begin(), u.end(), it) != 1) {
327  std::cerr << "No self-loops allowed, therefore (";
328  for (auto &it_print : u) {
329  std::cerr << it_print << ", ";

336  FloatType b = bias;
337 
338  if((vartype != Vartype::NONE) && (vartype != m_vartype)) {
339  if((m_vartype == Vartype::SPIN) && (vartype == Vartype::BINARY)) {
340  //convert from binary to spin
341  b /= 4;
342  add_offset(b);

349  b *= 4;

352  std::cerr << "Unknown vartype" << std::endl;
353  }
354  }
355  else {
356  //TODO(KSuzuki): Maybe bad code
357  for (auto &it : u) {
358  int flag = 0;
359  for (auto &it_polynomial: m_polynomial) {
360  if (it_polynomial.first.size() == 1 && it_polynomial.first[0] == it) {
361  flag = 1;
