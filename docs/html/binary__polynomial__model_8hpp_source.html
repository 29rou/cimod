329  void add_linear(const IndexType &v, const FloatType &bias, const Vartype vartype = Vartype::NONE) {
330 
331  //Check vartype when m_variables.empty() is true
332  if (m_variables.empty()) {
333  if (m_vartype == Vartype::NONE && vartype != Vartype::NONE) {
334  m_vartype = vartype;
335  }
336  else if (m_vartype == Vartype::NONE && vartype == Vartype::NONE) {
337  std::cerr << "Binary polynomial model is empty." << std::endl;
338  std::cerr << "Please set vartype to Vartype::SPIN or Vartype::BINARY" << std::endl;
339  return;
340  }
341  }
342 
343  //Check vartype
344  if ((vartype != Vartype::NONE) && (vartype != m_vartype)) {
345  if ((m_vartype == Vartype::SPIN) && (vartype == Vartype::BINARY)) {
346  std::cerr << "Cannot convert vartype=SPIN to vartype=BINARY" << std::endl;
347  return;
348  }
349  else if((m_vartype == Vartype::BINARY) && (vartype == Vartype::SPIN)) {
350  std::cerr << "Cannot convert vartype=BINARY to vartype=SPIN" << std::endl;
351  return;
352  }
353  else {
354  std::cerr << "Unknown vartype" << std::endl;
355  return;
356  }
357  }
358 
359  // Insert or assign the bias
360  FloatType value = 0;
361  std::vector<IndexType> index = {v};
362  if (m_polynomial.count(index) != 0) {
363  value = m_polynomial[index];
364  }
365  insert_or_assign(m_polynomial, index, value + bias);
366  m_variables.emplace(v);
367  }

374  void add_interaction(const std::vector<IndexType> &u, const FloatType &bias, const Vartype vartype = Vartype::NONE) {
375 
376  //Check the input interaction is valid
377  for (const auto &it : u) {
378  if (std::count(u.begin(), u.end(), it) != 1) {
379  /* Comment out in the case of IndexType = std::tuple for pybind11
380  std::cerr << "No self-loops allowed, therefore (";
381  for (const auto &it_print : u) {
382  std::cerr << it_print << ", ";
383  }
384  std::cerr << ") is not an allowed interaction" << std::endl;
385  */
386  std::cerr << "No self-loops allowed" << std::endl;
387  return;
388  }
389  }
390 
391  //Check vartype when m_variables.empty() is true
392  if (m_variables.empty()) {
393  if (m_vartype == Vartype::NONE && vartype != Vartype::NONE) {
394  m_vartype = vartype;
395  }
396  else if (m_vartype == Vartype::NONE && vartype == Vartype::NONE) {
397  std::cerr << "Binary polynomial model is empty." << std::endl;
398  std::cerr << "Please set vartype to Vartype::SPIN or Vartype::BINARY" << std::endl;
399  return;
400  }
401  }
402 
403  if((vartype != Vartype::NONE) && (vartype != m_vartype)) {
404  if((m_vartype == Vartype::SPIN) && (vartype == Vartype::BINARY)) {
405  std::cerr << "Cannot convert vartype=SPIN to vartype=BINARY" << std::endl;
406  return;
407  }
408  else if((m_vartype == Vartype::BINARY) && (vartype == Vartype::SPIN)) {
409  std::cerr << "Cannot convert vartype=BINARY to vartype=SPIN" << std::endl;
410  return;
411  }
412  else {
413  std::cerr << "Unknown vartype" << std::endl;
414  return;
415  }
416  }
417 
418  FloatType value = 0;
419  if (m_polynomial.count(u) != 0) {
420  value = m_polynomial[u];
421  }
422  insert_or_assign(m_polynomial, u, value + bias);
423 
424 
425  if (u.size() > 1) {
426  update_adjacency(u);
427  }
428 
429  for (const auto &it: u) {
430  if (m_variables.count(it) == 0) {
431  m_variables.emplace(it);
432  add_linear(it, 0);
433  }
434  }

507  void normalize(const std::pair<FloatType, FloatType> &bias_range = {1.0, 1.0},
508  const std::vector<std::vector<IndexType>> &ignored_variables = {}
509  ) {
510 
511  if (m_polynomial.empty()) {
512  return;
513  }
514 
515  auto comp = [](const auto &a, const auto &b) { return a.second < b.second; };
516  auto min_it = std::min_element(m_polynomial.begin(), m_polynomial.end(), comp);
517  auto max_it = std::max_element(m_polynomial.begin(), m_polynomial.end(), comp);
518 
519  auto min = min_it->second/bias_range.first;
520  auto max = max_it->second/bias_range.second;
521 
522  FloatType inv_scale = (min < max) ? max : min;
523 
524  if (inv_scale != 0.0) {
525  scale(1.0/inv_scale, ignored_variables);
526  }
527  }

541  FloatType energy(const Sample<IndexType> &sample) const {
542  FloatType en = 0.0;
543  for (const auto &it_polynomial: m_polynomial) {
544  int32_t multiple_variable = 1;
545  for (const auto &it_vec: it_polynomial.first) {
546  const auto v = sample.at(it_vec);
547  if (check_vartype(v, m_vartype)) {
548  multiple_variable *= v;
549  }
550  else {
551  multiple_variable = 0;
552  break;
553  }
554  }
555  en += static_cast<FloatType>(multiple_variable)*it_polynomial.second;
556  }
557  return en;
558  }

576  json to_serializable() const {
577  std::string schema_version = "3.0.0";
578  //set variables (sorted)
579  std::vector<IndexType> variables = _generate_indices();
580  size_t num_variables = variables.size();
581 
582  size_t num_interactions = m_polynomial.size();
583 
584  //set polynomial index and biases
585  std::vector<std::vector<size_t>> p_index;
586  std::vector<FloatType> p_bias;
587  for (const auto &it_polynomial: m_polynomial) {
588  p_index.push_back(std::vector<size_t>());
589  for (const auto &it_index: it_polynomial.first) {
590  auto it_ind = std::find(variables.begin(), variables.end(), it_index);
591  size_t idx = std::distance(variables.begin(), it_ind);
592  p_index[p_index.size() - 1].push_back(idx);
593  }
594  p_bias.push_back(it_polynomial.second);
595  }
596 
597  //set index_dtype
598  std::string index_dtype = num_variables <= 65536UL ? "uint16" : "uint32";
599 
600  //set bias_type
601  std::string bias_type;
602  if(typeid(FloatType) == typeid(float)) {
603  bias_type = "float32";
604  }
605  else if(typeid(FloatType) == typeid(double)) {
606  bias_type = "float64";
607  }
608  else {
609  std::cerr << "FloatType must be float or double." << std::endl;
610  }
611 
612  //set variable type
613  std::string variable_type;
614  if(m_vartype == Vartype::SPIN) {
615  variable_type = "SPIN";
616  }
617  else if(m_vartype == Vartype::BINARY) {
618  variable_type = "BINARY";
619  }
620  else {
621  std::cerr << "Variable type must be SPIN or BINARY." << std::endl;
622  }
623 
624  json output;
625  output["type"] = "BinaryPolynomialModel";
626  output["version"] = {{"bpm_schema", "3.0.0"}};
627  output["variable_labels"] = variables;
628  output["use_bytes"] = false;
629  output["index_type"] = index_dtype;
630  output["bias_type"] = bias_type;
631  output["num_variables"] = num_variables;
632  output["num_interactions"] = num_interactions;
633  output["variable_type"] = variable_type;
634  output["info"] = m_info;
635  output["polynomial_interactions"] = p_index;
636  output["polynomial_biases"] = p_bias;
637 
638  return output;
639  }

646  template <typename IndexType_serial = IndexType, typename FloatType_serial = FloatType>
647  static BinaryPolynomialModel<IndexType_serial, FloatType_serial> from_serializable(const json &input) {
648  //extract type and version
649  std::string type = input["type"];
650  if(type != "BinaryPolynomialModel") {
651  throw std::runtime_error("Type must be \"BinaryPolynomialModel\".\n");
652  }
653  std::string version = input["version"]["bpm_schema"];
654  if(version != "3.0.0") {
655  throw std::runtime_error("bpm_schema must be 3.0.0.\n");
656  }
657 
658  //extract variable_type
659  Vartype vartype;
660  std::string variable_type = input["variable_type"];
661  if(variable_type == "SPIN") {
662  vartype = Vartype::SPIN;
663  }
664  else if(variable_type == "BINARY") {
665  vartype = Vartype::BINARY;
666  }
667  else {
668  throw std::runtime_error("variable_type must be SPIN or BINARY.");
669  }
670 
671  //extract polynomial biases
672  std::vector<IndexType_serial> variables = input["variable_labels"];
673  std::vector<std::vector<size_t>> p_index = input["polynomial_interactions"];
674  std::vector<FloatType_serial> p_bias = input["polynomial_biases"];
675  Polynomial<IndexType_serial, FloatType_serial> polynomial;
676 
677  for (std::size_t i = 0; i < p_bias.size(); ++i) {
678  std::vector<IndexType_serial> temp_index;
679  for (std::size_t j = 0; j < p_index[i].size(); ++j) {
680  temp_index.push_back(variables[p_index[i][j]]);
681  }
682  if (i < p_index.size()) {
683  insert_or_assign(polynomial, temp_index, p_bias[i]);
684  }
685  }
686 
687  //extract info
688  std::string info = (input["info"].empty())?"":input["info"];
689 
690  //construct a BinaryPolynomialModel instance
691  return BinaryPolynomialModel<IndexType_serial, FloatType_serial>(polynomial, vartype, info);
692  }
