196  BinaryPolynomialModel(const std::vector<IndexType> &variables,
197  const PolynomialKeyList<std::size_t> &poly_key_distance_list,
198  const PolynomialValueList<FloatType> &poly_value_list,
199  const Vartype vartype
200  ): vartype_(vartype) {
201 
202  if (poly_key_distance_list.size() != poly_value_list.size()) {
203  throw std::runtime_error("The sizes of key_list and value_list must match each other");
204  }
205 
206  variables_ = std::unordered_set<IndexType>(variables.begin(), variables.end());
207 
208  if (variables_.size() != variables.size()) {
209  throw std::runtime_error("Unknown error. It seems that the input variables contain the same variables");
210  }
211 
212  std::size_t num_interactions = poly_key_distance_list.size();
213  poly_key_list_.resize(num_interactions);
214  poly_value_list_.resize(num_interactions);
215 
216 #pragma omp parallel for
217  for (std::size_t i = 0; i < num_interactions; ++i) {
218  std::vector<IndexType> temp;
219  for (const auto &it: poly_key_distance_list[i]) {
220  temp.push_back(variables[it]);
221  }
222  std::sort(temp.begin(), temp.end());
223  poly_key_list_[i] = temp;
224  poly_value_list_[i] = poly_value_list[i];
225  }
226 
227  for (std::size_t i = 0; i < num_interactions; ++i) {
228  poly_key_inv_[poly_key_list_[i]] = i;
229  for (const auto &it: poly_key_list_[i]) {
230  each_variable_num_[it]++;
231  }
232  }

348  std::sort(key.begin(), key.end());
349  if (poly_key_inv_.count(key) == 0) {
350  return;
351  }
352 
353  for (const auto &index: key) {
354  if (each_variable_num_[index] >= 2) {
355  each_variable_num_[index]--;

426  void add_interaction(std::vector<IndexType> &key, const FloatType &value, const Vartype vartype = Vartype::NONE) {
427  if (std::abs(value) <= 0.0) {
428  return;
429  }
430  std::sort(key.begin(), key.end());
431  CheckKeySelfLoop(key);
432  if (vartype_ == vartype || vartype == Vartype::NONE) {
433  SetKeyAndValue(key, value);
434  }
435  else {
436  const std::size_t original_key_size = key.size();
437  const std::size_t changed_key_list_size = IntegerPower(2, original_key_size);
438 
439  if (vartype_ == Vartype::SPIN && vartype == Vartype::BINARY) {
440  for (std::size_t i = 0; i < changed_key_list_size; ++i) {
441  const auto changed_key = GenerateChangedKey(key, i);
442  int sign = ((original_key_size - changed_key.size())%2 == 0) ? 1.0 : -1.0;
443  SetKeyAndValue(changed_key, value*IntegerPower(2, changed_key.size())*sign);
444  }
445  }
446  else if (vartype_ == Vartype::BINARY && vartype == Vartype::SPIN) {
447  FloatType changed_value = value*(1.0/changed_key_list_size);
448  for (std::size_t i = 0; i < changed_key_list_size; ++i) {
449  SetKeyAndValue(GenerateChangedKey(key, i), changed_value);
450  }
451  }
452  else {
453  throw std::runtime_error("Unknown vartype error");
454  }
455  }
456  }

514  FloatType energy(const Sample<IndexType> &sample, bool omp_flag = true) const {
515  if (sample.size() != get_num_variables()) {
516  throw std::runtime_error("The size of sample must be equal to num_variables");
517  }
518 
519  if (get_num_interactions() == 0) {
520  return 0.0;
521  }
522 
523  std::size_t num_interactions = get_num_interactions();
524  FloatType val = 0.0;
525 
526  if (omp_flag) {
527 #pragma omp parallel for reduction (+: val)
528  for (std::size_t i = 0; i < num_interactions; ++i) {
529  int32_t spin_multiple = 1;
530  for (const auto &index: poly_key_list_[i]) {
531  spin_multiple *= sample.at(index);
532  if (spin_multiple == 0.0) {
533  break;
534  }
535  }
536  val += spin_multiple*poly_value_list_[i];
537  }
538  }
539  else {
540  for (std::size_t i = 0; i < num_interactions; ++i) {
541  int32_t spin_multiple = 1;
542  for (const auto &index: poly_key_list_[i]) {
543  spin_multiple *= sample.at(index);
544  if (spin_multiple == 0.0) {
545  break;
546  }
547  }
548  val += spin_multiple*poly_value_list_[i];
549  }
550  }
551  return val;
552  }

570  void scale(const FloatType scalar,
571  const PolynomialKeyList<IndexType> &ignored_interactions = {},
572  const bool ignored_offset = false) {
573 
574  std::size_t num_interactions = get_num_interactions();
575 
576  for (std::size_t i = 0; i < num_interactions; ++i) {
577  poly_value_list_[i] *= scalar;
578  }
579 
580  FloatType scalar_inv = 1.0/scalar;
581  for (const auto &key: ignored_interactions) {
582  if (poly_key_inv_.count(key) != 0) {
583  poly_value_list_[poly_key_inv_[key]] *= scalar_inv;
584  }
585  }
586 
587  if (ignored_offset == true && poly_key_inv_.count(std::vector<IndexType>{}) != 0 && std::count(ignored_interactions.begin(), ignored_interactions.end(), std::vector<IndexType>{}) == 0) {
588  poly_value_list_[poly_key_inv_[std::vector<IndexType>{}]] *= scalar_inv;
589  }
590  }

596  void normalize(const std::pair<FloatType, FloatType> &range = {1.0, 1.0},
597  const PolynomialKeyList<IndexType> &ignored_interactions = {},
598  const bool ignored_offset = false) {
599 
600  if (get_num_interactions() == 0) {
601  return;
602  }
603 
604  FloatType max_poly_value = poly_value_list_[0];
605  FloatType min_poly_value = poly_value_list_[0];
606 
607  for (const auto &poly_value: poly_value_list_) {
608  if (max_poly_value < poly_value) {
609  max_poly_value = poly_value;
610  }
611  if (min_poly_value > poly_value) {
612  min_poly_value = poly_value;
613  }
614  }
615 
616  FloatType inv_scale = std::max(min_poly_value/range.first, max_poly_value/range.second);
617 
618  if (inv_scale != 0.0) {
619  scale(1.0/inv_scale, ignored_interactions, ignored_offset);
620  }
621  }

667  Polynomial<IndexType, FloatType> to_hubo() const {
668  if (vartype_ == Vartype::BINARY) {
669  return get_polynomial();
670  }
671  Polynomial<IndexType, FloatType> poly_map;
672  std::size_t num_interactions = get_num_interactions();
673  for (std::size_t i = 0; i < num_interactions; ++i) {
674  const std::vector<IndexType> &original_key = poly_key_list_[i];
675  const FloatType original_value = poly_value_list_[i];
676  const std::size_t original_key_size = original_key.size();
677  const std::size_t changed_key_list_size = IntegerPower(2, original_key_size);
678 
679  for (std::size_t j = 0; j < changed_key_list_size; ++j) {
680  const auto changed_key = GenerateChangedKey(original_key, j);
681  int sign = ((original_key_size - changed_key.size())%2 == 0) ? 1.0 : -1.0;
682  FloatType changed_value = original_value*IntegerPower(2, changed_key.size())*sign;
683  poly_map[changed_key] += changed_value;
684  if (poly_map[changed_key] == 0.0) {

861  void SetKeyAndValue(const std::vector<IndexType> &key, const FloatType &value) {
862  //key is assumed to be sorted
863  if (poly_key_inv_.count(key) == 0) {
864  poly_key_inv_[key] = poly_value_list_.size();
865  poly_key_list_.push_back(key);
866  poly_value_list_.push_back(value);

870  remove_interaction(key);

872  }
873  poly_value_list_[poly_key_inv_[key]] += value;
874  }

878  }

897  std::vector<IndexType> GenerateChangedKey(const std::vector<IndexType> &original_key, const std::size_t num_of_key) const {
898  if (original_key.size() >= UINT16_MAX) {
899  throw std::runtime_error("Too large degree of the interaction");
900  }
901  const std::size_t original_key_size = original_key.size();
902  std::bitset<UINT16_MAX> bs(num_of_key);
903  std::vector<IndexType> changed_key;
904  for (std::size_t i = 0; i < original_key_size; ++i) {
905  if (bs[i]) {
906  changed_key.push_back(original_key[i]);
907  }
908  }

915  if (vartype_ == Vartype::SPIN) {
916  return *this;
917  }

920  PolynomialValueList<FloatType> new_value_list;

922 
923  for (std::size_t i = 0; i < num_interactions; ++i) {
924  const std::vector<IndexType> &original_key = poly_key_list_[i];
925  const FloatType original_value = poly_value_list_[i];
926  const std::size_t original_key_size = original_key.size();
927  const std::size_t changed_key_list_size = IntegerPower(2, original_key_size);
928 
929  FloatType changed_value = original_value*(1.0/changed_key_list_size);
930 
931  for (std::size_t j = 0; j < changed_key_list_size; ++j) {
932  new_key_list.push_back(GenerateChangedKey(original_key , j));
933  new_value_list.push_back(changed_value);
934  }
935  }

937  }

953  const FloatType original_value = poly_value_list_[i];
954  const std::size_t original_key_size = original_key.size();
955  const std::size_t changed_key_list_size = IntegerPower(2, original_key_size);
956

961  new_value_list.push_back(original_value*IntegerPower(2, changed_key.size())*sign);
962  }
963  }
964  return BinaryPolynomialModel(new_key_list, new_value_list, Vartype::BINARY);
