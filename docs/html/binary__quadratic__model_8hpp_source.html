456  void add_variable
457  (
458  const IndexType &v,
459  const FloatType &bias,
460  const Vartype vartype = Vartype::NONE
461  )
462  {
463  FloatType b = bias;
464 
465  // handle the case that a different vartype is provided
466  if((vartype!=Vartype::NONE)&&(vartype!=m_vartype))
467  {
468  if((m_vartype == Vartype::SPIN)&&(vartype == Vartype::BINARY))
469  {
470  b /= 2;
471  m_offset += b;
472  }
473  else if((m_vartype == Vartype::BINARY)&&(vartype == Vartype::SPIN))
474  {
475  m_offset -= b;
476  b *= 2;
477  }
478  else
479  {
480  std::cerr << "Unknown vartype" << std::endl;
481  }
482  }
483 
484  // Insert or assign the bias
485  FloatType value = 0;
486  if(m_linear.count(v) != 0)
487  {
488  value = m_linear[v];
489  }
490  insert_or_assign(m_linear, v, value + b);
491  };

519  void add_interaction
520  (
521  const IndexType &u,
522  const IndexType &v,
523  const FloatType &bias,
524  const Vartype vartype = Vartype::NONE
525  )
526  {
527  if(u == v)
528  {
529  std::cerr << "No self-loops allowed" << std::endl;
530  }
531  else
532  {
533  FloatType b = bias;
534  if((vartype!=Vartype::NONE)&&(vartype!=m_vartype))
535  {
536  if((m_vartype == Vartype::SPIN)&&(vartype == Vartype::BINARY))
537  {
538  //convert from binary to spin
539  b /= 4;
540  add_offset(b);
541  add_variable(u, b);
542  add_variable(v, b);
543  }
544  else if((m_vartype == Vartype::BINARY)&&(vartype == Vartype::SPIN))
545  {
546  //convert from spin to binary
547  add_offset(b);
548  add_variable(u, -2 * b);
549  add_variable(v, -2 * b);
550  b *= 4;
551  }
552  else
553  {
554  std::cerr << "Unknown vartype" << std::endl;
555  }
556  }
557  else
558  {
559  if(m_linear.count(u) == 0)
560  {
561  add_variable(u, 0);
562  }
563  if(m_linear.count(v) == 0)
564  {
565  add_variable(v, 0);
566  }
567  }
568 
569  FloatType value = 0;
570  std::pair<IndexType, IndexType> p1 = std::make_pair(u, v);
571  if(m_quadratic.count(p1) != 0)
572  {
573  value = m_quadratic[p1];
574  }
575  insert_or_assign(m_quadratic, p1, value + b);
576  update_adjacency(u, v);
577  }
578  };

704  void scale
705  (
706  const FloatType &scalar,
707  const std::vector<IndexType> &ignored_variables = {},
708  const std::vector<std::pair<IndexType, IndexType>> &ignored_interactions = {},
709  const bool ignored_offset = false
710  )
711  {
712  // scaling linear
713  for(auto &it : m_linear)
714  {
715  if(std::find(ignored_variables.begin(), ignored_variables.end(), it.first) != ignored_variables.end() || ignored_variables.empty())
716  {
717  it.second *= scalar;
718  }
719  }
720 
721  // scaling quadratic
722  for(auto &it : m_quadratic)
723  {
724  if(std::find(ignored_interactions.begin(), ignored_interactions.end(), it.first) != ignored_interactions.end() || ignored_variables.empty())
725  {
726  it.second *= scalar;
727  }
728  }
729 
730  // scaling offset
731  if(!ignored_offset)
732  {
733  m_offset *= scalar;
734  }
735  };

748  void normalize
749  (
750  const std::pair<FloatType, FloatType> &bias_range = {1.0, 1.0},
751  const bool use_quadratic_range = false,
752  const std::pair<FloatType, FloatType> &quadratic_range = {1.0, 1.0},
753  const std::vector<IndexType> &ignored_variables = {},
754  const std::vector<std::pair<IndexType, IndexType>> &ignored_interactions = {},
755  const bool ignored_offset = false
756  )
757  {
758  // parse range
759  std::pair<FloatType, FloatType> l_range = bias_range;
760  std::pair<FloatType, FloatType> q_range;
761  if(!use_quadratic_range)
762  {
763  q_range = bias_range;
764  }
765  else
766  {
767  q_range = quadratic_range;
768  }
769 
770  // calculate scaling value
771  auto comp = [](const auto &a, const auto &b) { return a.second < b.second; };
772  auto it_lin_min = std::min_element(m_linear.begin(), m_linear.end(), comp);
773  auto it_lin_max = std::max_element(m_linear.begin(), m_linear.end(), comp);
774  auto it_quad_min = std::min_element(m_quadratic.begin(), m_quadratic.end(), comp);
775  auto it_quad_max = std::max_element(m_quadratic.begin(), m_quadratic.end(), comp);
776 
777  std::vector<FloatType> v_scale =
778  {
779  it_lin_min->second / l_range.first,
780  it_lin_max->second / l_range.second,
781  it_quad_min->second / q_range.first,
782  it_quad_max->second / q_range.second
783  };
784  FloatType inv_scale = *std::max_element(v_scale.begin(), v_scale.end());
785 
786  // scaling
787  if(inv_scale != 0.0)
788  {
789  scale(1.0 / inv_scale, ignored_variables, ignored_interactions, ignored_offset);
790  }
791  };

845  void flip_variable
846  (
847  const IndexType &v
848  )
849  {
850  // check variable
851  if(m_linear.count(v)==0)
852  {
853  std::cerr << "not a variable in the binary quadratic model." << std::endl;
854  return;
855  }
856 
857  if(m_vartype == Vartype::SPIN)
858  {
859  m_linear[v] *= -1.0;
860  for(auto &it : m_quadratic)
861  {
862  if(it.first.first == v || it.first.second == v)
863  {
864  it.second *= -1.0;
865  update_adjacency(it.first.first, it.first.second);
866  }
867  }
868  }
869  else if(m_vartype == Vartype::BINARY)
870  {
871  add_offset(m_linear[v]);
872  m_linear[v] *= -1.0;
873 
874  for(auto &it : m_quadratic)
875  {
876  if(it.first.first == v)
877  {
878  m_linear[it.first.second] += it.second;
879  it.second *= -1.0;
880  update_adjacency(it.first.first, it.first.second);
881  }
882  else if(it.first.second == v)
883  {
884  m_linear[it.first.first] += it.second;
885  it.second *= -1.0;
886  update_adjacency(it.first.first, it.first.second);
887  }
888  }
889  }

913  void contract_variables
914  (
915  const IndexType &u,
916  const IndexType &v
917  )
918  {
919  // check variable
920  if(m_linear.count(v)==0)
921  {
922  std::cerr << "not a variable in the binary quadratic model." << std::endl;
923  return;
924  }
925  if(m_linear.count(u)==0)
926  {
927  std::cerr << "not a variable in the binary quadratic model." << std::endl;
928  return;
929  }
930 
931  auto p1 = std::make_pair(u, v);
932  auto p2 = std::make_pair(v, u);
933  if(m_quadratic.count(p1) != 0)
934  {
935  if(m_vartype == Vartype::BINARY)
936  {
937  add_variable(u, m_quadratic[p1]);
938  }
939  else if(m_vartype == Vartype::SPIN)
940  {
941  add_offset(m_quadratic[p1]);
942  }
943  remove_interaction(u, v);
944  }
945  if(m_quadratic.count(p2) != 0)
946  {
947  if(m_vartype == Vartype::BINARY)
948  {
949  add_variable(u, m_quadratic[p2]);
950  }
951  else if(m_vartype == Vartype::SPIN)
952  {
953  add_offset(m_quadratic[p2]);
954  }
955  remove_interaction(v, u);
956  }
957 
958  std::vector<std::pair<IndexType, IndexType>> interactions;
959  for(auto &it : m_quadratic)
960  {
961  if(it.first.first == v)
962  {
963  add_interaction(u, it.first.second, it.second);
964  update_adjacency(u, it.first.second);
965  interactions.push_back(it.first);
966  }
967  else if(it.first.second == v)
968  {
969  add_interaction(it.first.first, u, it.second);
970  update_adjacency(it.first.first, u);
971  interactions.push_back(it.first);
972  }
973  }
974  remove_interactions_from(interactions);
975 
976  add_variable(u, m_linear[v]);
977  remove_variable(v);
978  };

989  BinaryQuadraticModel change_vartype
990  (
991  const Vartype &vartype,
992  bool inplace=true
993  )
994  {
995  Linear<IndexType, FloatType> linear;
996  Quadratic<IndexType, FloatType> quadratic;
997  FloatType offset = 0.0;
998 
999  if(m_vartype == Vartype::BINARY && vartype == Vartype::SPIN) // binary -> spin
1000  {
1001  std::tie(linear, quadratic, offset) = binary_to_spin(m_linear, m_quadratic, m_offset);
1002  }
1003  else if(m_vartype == Vartype::SPIN && vartype == Vartype::BINARY) // spin -> binary
1004  {
1005  std::tie(linear, quadratic, offset) = spin_to_binary(m_linear, m_quadratic, m_offset);
1006  }
1007  else
1008  {
1009  std::tie(linear, quadratic, offset) = std::tie(m_linear, m_quadratic, m_offset);
1010  }
1011 
1012  BinaryQuadraticModel bqm(linear, quadratic, offset, vartype, m_info);
1013 
1014  if(inplace == true)
1015  {
1016  //inplace
1017  m_linear = bqm.get_linear();
1018  m_quadratic = bqm.get_quadratic();
1019  m_offset = bqm.get_offset();
1020  m_adj = bqm.get_adjacency();
1021  m_info = bqm.get_info();
1022  m_vartype = bqm.get_vartype();
1023  }
1024 
1025  return bqm;
1026  };

1038  static std::tuple<Linear<IndexType, FloatType>, Quadratic<IndexType, FloatType>, FloatType> spin_to_binary
1039  (
1040  const Linear<IndexType, FloatType> &linear,
1041  const Quadratic<IndexType, FloatType> &quadratic,
1042  const FloatType &offset
1043  )
1044  {
1045  Linear<IndexType, FloatType> new_linear;
1046  Quadratic<IndexType, FloatType> new_quadratic;
1047  FloatType new_offset = offset;
1048  FloatType linear_offset = 0.0;
1049  FloatType quadratic_offset = 0.0;
1050 
1051  for(auto &it : linear)
1052  {
1053  insert_or_assign(new_linear, it.first, static_cast<FloatType>(2.0 * it.second));
1054  linear_offset += it.second;
1055  }
1056 
1057  for(auto &it : quadratic)
1058  {
1059  insert_or_assign(new_quadratic, it.first, static_cast<FloatType>(4.0 * it.second));
1060  new_linear[it.first.first] -= 2.0 * it.second;
1061  new_linear[it.first.second] -= 2.0 * it.second;
1062  quadratic_offset += it.second;
1063  }
1064 
1065  new_offset += quadratic_offset - linear_offset;
1066 
1067  return std::make_tuple(new_linear, new_quadratic, new_offset);
1068  };
